
            Dictionary<string, List<string>> unqualifiedToQualified = new Dictionary<string, List<string>>();
            foreach (Model table in Schema.Singleton.Models) {
                string unqualified = table.UnqualifiedClassName;

                List<string> qualifieds;
                if (!unqualifiedToQualified.TryGetValue(unqualified, out qualifieds)) {
                    qualifieds = new List<string>();
                    unqualifiedToQualified[unqualified] = qualifieds;
                }
                qualifieds.Add(table.ClassName);
            }

            Console.WriteLine("Name Clashes...");
            foreach (List<string> qualifieds in unqualifiedToQualified.Values.Where(x => x.Count > 1))
                Console.WriteLine(string.Join(", ", qualifieds));
